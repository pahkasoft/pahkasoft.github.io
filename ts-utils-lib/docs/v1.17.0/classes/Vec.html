    Class Vec

    Vector class.

    // Example usage:
    const a = new Vec(1, 2);
    const b = new Vec(3, 4);

    console.log(a.add(b).toString());     // Vec(4, 6)
    console.log(a.add(3, 3).toString());  // Vec(4, 5)
    console.log(a.mul(2).toString());     // Vec(2, 4)
    console.log(a.equals(b));             // false
    console.log(a.clone().equals(a));     // true
